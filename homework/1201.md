int BinarySearch(int arr[], int low, int high, int key) {<br>
    if (low > high) {<br>
        return -1; // 查找失败<br>
    }<br>
    int mid = (low + high) / 2; <br>
    if (arr[mid] == key) {<br>
        return mid; // 查找成功<br>
    } else if (arr[mid] > key) {<br>
        return BinarySearch(arr, low, mid - 1, key); // 在左半部分查找<br>
    } else {<br>
        return BinarySearch(arr, mid + 1, high, key); // 在右半部分查找<br>
    }
}
