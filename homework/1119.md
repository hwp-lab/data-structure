void DFS_NonRecursive(ALGraph *G, int v) {                                  <br>
    if (v < 0 || v >= G->vexnum) {<br>
        printf("顶点索引错误！\n");<br>
        return;<br>
    }<br>
    <br>
    bool visited[MAX_VERTEX_NUM] = {false}; // 访问标记数组<br>
    Stack stack;<br>
    initStack(&stack);<br>
    <br>
    printf("从顶点 %c 开始的深度优先遍历（非递归）: ", G->vertices[v].data);<br>
    <br>
    // 起始顶点入栈并标记为已访问<br>
    push(&stack, v);<br>
    visited[v] = true;<br>
    printf("%c ", G->vertices[v].data);<br>
    <br>
    while (!isStackEmpty(&stack)) {<br>
        int current;<br>
        getTop(&stack, &current); // 查看栈顶元素，不出栈<br>
        <br>
        // 查找当前顶点的第一个未访问邻接点<br>
        ArcNode *p = G->vertices[current].firstarc;<br>
        int nextVertex = -1;<br>
        <br>
        while (p != NULL) {<br>
            if (!visited[p->adjvex]) {<br>
                nextVertex = p->adjvex;<br>
                break;<br>
            }<br>
            p = p->nextarc;<br>
        }<br>
        <br>
        if (nextVertex != -1) {<br>
            // 找到未访问的邻接点，访问并入栈<br>
            visited[nextVertex] = true;<br>
            push(&stack, nextVertex);<br>
            printf("%c ", G->vertices[nextVertex].data);<br>
        } else {<br>
            // 当前顶点的所有邻接点都已访问，出栈<br>
            pop(&stack, &current);<br>
        }<br>
    }<br>
    printf("\n");<br>
}<br>
