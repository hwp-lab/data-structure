/*试编写算法，对一棵以孩子-兄弟链表表示的树统计叶子的个数<br>
对以孩子-兄弟链表表示的树编写计算树的深度的算法*/<br>
#ifdef c_1105_//宏定义不可以数字开头<br>
#define c_1105_<br>
<br>
#include<stdio.h><br>
<br>
typedef char Elemtype；<br>
<br>
typedef struct node{<br>
    Elemtype data;<br>
    struct node *firstchild,*nextsibling;<br>
}CSnode;<br>
<br>
int getleafNum(CSnode *node){<br>
    if(node == NULL)<br>
        return 0;<br>
    if(node->firstchild == NULL)<br>
        return 1 + getleafNum(node->nextsibling);<br>
    else                                                                        <br>
        return getleafNum(node->firstchild) + getleafNum(node->nextsibling);<br>
}<br>
<br>
//求深度<br>
int getDepth(CSnode *node) {<br>
    if (node == NULL)<br>
        return 0;<br>
    <br>
    int childDepth = getDepth(node->firstchild);  // 孩子子树深度<br>
    int siblingDepth = getDepth(node->nextsibling);  // 兄弟子树深度<br>
    <br>
    if(childDepth + 1 > siblingDepth) <br>
        return childDepth + 1;<br>
    else <br>
        return siblingDepth;<br>
}<br>
#endif<br>
