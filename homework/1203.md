typedef struct Node                                                     <br>
{<br>
    int data;<br>
    int count;<br>
    struct Node *llink;<br>
    struct Node *rlink;<br>
}Node,*Tree;<br>
<br>
Node *SearchInsert(Tree *root,int x){<br>
    if (*root == NULL) {<br>
        Node* newNode = (Node*)malloc(sizeof(Node));<br>
        newNode->data = x;<br>
        newNode->count = 1;<br>
        newNode->llink = NULL;<br>
        newNode->rlink = NULL;<br>
        *root = newNode;<br>
        return newNode;<br>
    }<br>
<br>
    Node *p=*root;//Node *p; p=*root<br>
    Node *parent=NULL;<br>
<br>
    while(p!=NULL){<br>
        if(p->data==x){<br>
            p->count++;<br>
            return p;<br>
        }<br>
        parent=p;                                   <br>
        if(p->data>x)<br>
            p=p->llink;<br>
        else<br>
            p=p->rlink;<br>
    }<br>
<br>
    Node* newNode=(Node*)malloc(sizeof(Node));<br>
    newNode->count=1;<br>
    newNode->data=x;<br>
    newNode->llink=NULL;<br>
    newNode->rlink=NULL;<br>
<br>
    if(parent->data>x)<br>
        parent->llink=newNode;<br>
    else<br>
        parent->rlink=newNode;<br>
    <br>
    return newNode;<br>
}
