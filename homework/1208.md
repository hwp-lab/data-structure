typedef struct AVLNode {                                <br>
    int key;<br>
    int b;  // 平衡因子 = 右子树高度 - 左子树高度<br>
    struct AVLNode *lchild, *rchild;<br>
} AVLNode, *AVLTree;<br>
<br>
// 算法：利用平衡因子求平衡二叉树的高度<br>
int getAVLHeight(AVLTree T) {<br>
    if (T == NULL) return 0;  // 空树高度为0<br>
    <br>
    int height = 1;  // 当前路径高度<br>
    AVLNode *p = T;<br>
    <br>
    while (p != NULL) {<br>
        if (p->b >= 0) {<br>
            // 右子树高度 >= 左子树高度<br>
            if (p->rchild != NULL) {<br>
                height++;<br>
                p = p->rchild;<br>
            } else {<br>
                break;  // 到达叶子节点<br>
            }<br>
        } else {<br>
            // 左子树高度 > 右子树高度<br>
            if (p->lchild != NULL) {<br>
                height++;<br>
                p = p->lchild;<br>
            } else {<br>
                break;  // 到达叶子节点<br>
            }<br>
        }<br>
    }<br>
    <br>
    return height;<br>
}<br>

// 辅助函数：创建新结点<br>
AVLNode* createNode(int key, int b) {<br>
    AVLNode *node = (AVLNode*)malloc(sizeof(AVLNode));<br>
    node->key = key;<br>
    node->b = b;<br>
    node->lchild = NULL;<br>
    node->rchild = NULL;<br>
    return node;<br>
}<br>
